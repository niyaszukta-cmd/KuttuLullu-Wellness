# ── Data helpers ──────────────────────────────────────────────────────────────
def load_json(path, default):
    if os.path.exists(path):
        with open(path) as f:
            return json.load(f)
    return default

def save_json(path, data):
    with open(path, "w") as f:
        json.dump(data, f, indent=2, default=str)

def get_habits():
    return load_json(HABITS_FILE, [])

def save_habits(habits):
    save_json(HABITS_FILE, habits)

def get_logs():
    return load_json(LOGS_FILE, {})

def save_logs(logs):
    save_json(LOGS_FILE, logs)

def today_str():
    return date.today().isoformat()

def toggle_log(habit_id: str):
    logs = get_logs()
    td = today_str()
    key = f"{habit_id}_{td}"
    logs[key] = not logs.get(key, False)
    save_logs(logs)

def is_done_today(habit_id: str) -> bool:
    logs = get_logs()
    return logs.get(f"{habit_id}_{today_str()}", False)

def compute_streak(habit_id: str, logs: dict) -> int:
    streak = 0
    d = date.today()
    while True:
        key = f"{habit_id}_{d.isoformat()}"
        if logs.get(key, False):
            streak += 1
            d -= timedelta(days=1)
        else:
            break
    return streak

def compute_completion_rate(habit_id: str, logs: dict, days: int = 30) -> float:
    count = 0
    for i in range(days):
        d = date.today() - timedelta(days=i)
        if logs.get(f"{habit_id}_{d.isoformat()}", False):
            count += 1
    return count / days * 100

def get_badge(streak: int) -> str:
    badge = ""
    for threshold, name in sorted(BADGE_THRESHOLDS.items(), reverse=True):
        if streak >= threshold:
            badge = name
            break
    return badge

def heatmap_html(habit_id: str, logs: dict, weeks: int = 16) -> str:
    cells = []
    today = date.today()
    start = today - timedelta(weeks=weeks)
    d = start
    while d <= today:
        key = f"{habit_id}_{d.isoformat()}"
        done = logs.get(key, False)
        if done:
            color = "#6366f1"
            opacity = "1"
        else:
            color = "#1e1e2e"
            opacity = "1"
        tip = f"{d.isoformat()} {'✓' if done else '○'}"
        cells.append(
            f'<span class="heatmap-cell" style="background:{color};opacity:{opacity}" title="{tip}"></span>'
        )
        d += timedelta(days=1)
    return f'<div class="calendar-grid">{"".join(cells)}</div>'
